Plug 'vimwiki/vimwiki' " wiki for vim
if executable(expand('~/.venv/bin/vimwiki_markdown'))
  let g:vimwiki_custom_wiki2html = expand('~/.venv/bin/vimwiki_markdown')
endif
if executable('vimwiki_markdown')
  let g:vimwiki_custom_wiki2html = 'vimwiki_markdown'
endif
let g:vimwiki_syntax = 'markdown'
let g:vimwiki_links_space_char = '_'
let g:vimwiki_dir_link = 'index'
let g:vimwiki_ext = '.md'
let g:vimwiki_automatic_nested_syntaxes = 1
let g:vimwiki_auto_toc = 1
let g:vimwiki_diary_rel_path = 'dairy_'.hostname()
let g:vimwiki_template_path = '~/project/wiki/template/'
let g:vimwiki_list = [
      \ {'name': '/', 'path': '~/project/wiki/docs/'},
      \ {'name': 'context', 'path': '~/project/wiki/docs/context/'},
      \ {'name': 'office', 'path': '~/project/wiki/docs/context/office/'},
      \ {'name': 'project', 'path': '~/project/wiki/docs/context/project/'},
      \ {'name': 'language', 'path': '~/project/wiki/docs/language/'},
      \ {'name': 'python', 'path': '~/project/wiki/docs/language/py/'},
      \ {'name': 'rust', 'path': '~/project/wiki/docs/language/rs/'},
      \ {'name': 'library', 'path': '~/project/wiki/docs/library/'},
      \ {'name': 'software', 'path': '~/project/wiki/docs/software/'},
      \ {'name': 'linux', 'path': '~/project/wiki/docs/software/linux/'},
      \ {'name': 'vim', 'path': '~/project/wiki/docs/software/vim/'},
      \ ]
